```kotlin

    @Test

        // TODO
    }

```

[![](https://jitpack.io/v/langara/RxMock.svg)](https://jitpack.io/#langara/RxMock)

### Building with JitPack
```gradle
    repositories {
        maven { url "https://jitpack.io" }
    }
   
    dependencies {
        testImplementation 'com.github.langara:RxMock:master-SNAPSHOT'
    }
```
